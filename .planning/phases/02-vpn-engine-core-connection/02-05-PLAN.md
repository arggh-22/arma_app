5. Handle tap:
```dart
void _handleTap() {
  switch (status) {
    case Disconnected():
      if (activeServer != null) {
        ref.read(connectionProvider.notifier).connect(activeServer);
      }
    case Connected():
      ref.read(connectionProvider.notifier).disconnect();
    case Connecting():
    case Disconnecting():
      break;  // Ignore taps during transition
  }
}
```

6. Build the button widget:
```dart
Semantics(
  label: semanticLabel,
  child: GestureDetector(
    onTap: (status is Connecting || status is Disconnecting) ? null : _handleTap,
    child: AnimatedContainer(
      duration: const Duration(milliseconds: 300),
      width: 120,
      height: 120,
      decoration: BoxDecoration(
        shape: BoxShape.circle,
        color: buttonColor,
        boxShadow: glowColor != null ? [
          BoxShadow(color: glowColor, blurRadius: 20, spreadRadius: 5),
        ] : null,

      child: const Icon(Icons.power_settings_new, color: Colors.white, size: 48),

    .animate(target: isAnimating ? 1 : 0)

      end: const Offset(1.05, 1.05),

      curve: Curves.easeInOut,

      end: const Offset(1.0, 1.0),
      duration: 800.ms,
      curve: Curves.easeInOut,
    )
    .animate(onPlay: isAnimating ? (c) => c.repeat() : null)
```

- **Disconnecting**: static grey, no animation

`_handleStateChange(ConnectionStatus status)`:
- If `Connected`: start timer that increments `_elapsed` every second, calculate initial elapsed from `connectedAt`
- If `Disconnected` or `Disconnecting`: cancel timer, reset `_elapsed = Duration.zero`

```dart
void _handleStateChange(ConnectionStatus status) {
  switch (status) {
    case Connected(:final connectedAt):
      _elapsed = DateTime.now().difference(connectedAt);
      _timer?.cancel();
      _timer = Timer.periodic(const Duration(seconds: 1), (_) {
        setState(() {
          _elapsed = DateTime.now().difference(connectedAt);
        });
      });
    case Disconnected():
    case Disconnecting():
      _timer?.cancel();
      setState(() { _elapsed = Duration.zero; });
    case Connecting():
      break;
  }
}
```
