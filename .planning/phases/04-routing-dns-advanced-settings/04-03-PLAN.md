"getInstalledApps" -> {
    CoroutineScope(Dispatchers.IO).launch {
        try {
            val pm = applicationContext.packageManager
            val apps = pm.getInstalledApplications(0)
                .filter { it.flags and ApplicationInfo.FLAG_SYSTEM == 0 }
                .map { appInfo ->
                    val iconBase64 = try {
                        val drawable = pm.getApplicationIcon(appInfo)
                        val bitmap = if (drawable is BitmapDrawable) {
                            drawable.bitmap
                        } else {
                            val bmp = Bitmap.createBitmap(48, 48, Bitmap.Config.ARGB_8888)
                            val canvas = Canvas(bmp)
                            drawable.setBounds(0, 0, 48, 48)
                            drawable.draw(canvas)
                            bmp
                        }
                        val stream = ByteArrayOutputStream()
                        bitmap.compress(Bitmap.CompressFormat.PNG, 80, stream)
                        Base64.encodeToString(stream.toByteArray(), Base64.NO_WRAP)
                    } catch (e: Exception) {
                        ""
                    }
                    mapOf(
                        "packageName" to appInfo.packageName,
                        "appName" to (pm.getApplicationLabel(appInfo)?.toString() ?: appInfo.packageName),
                        "icon" to iconBase64
                    )
                }
                .sortedBy { (it["appName"] as String).lowercase() }
            withContext(Dispatchers.Main) {
                result.success(apps)
            }
        } catch (e: Exception) {
            withContext(Dispatchers.Main) {
                result.error("APP_LIST_FAILED", e.message, null)
            }
        }
    }
}
"setPerAppConfig" -> {
    val mode = call.argument<String>("mode") // "blacklist", "whitelist", or null
    val apps = call.argument<List<String>>("selectedApps") ?: emptyList()
    // Store in SharedPreferences so VPN service can read
    val prefs = getSharedPreferences("per_app_config", MODE_PRIVATE)
    prefs.edit()
        .putString("per_app_mode", mode)
        .putStringSet("selected_apps", apps.toSet())
        .apply()
    Log.w(TAG, "setPerAppConfig: mode=$mode, apps=${apps.size}")
    result.success(true)
}

The per-app config is stored in a separate SharedPreferences file (`per_app_config`) because the VPN service runs in a separate process (`:vpn_process`) and cannot access the main Flutter SharedPreferences directly. Using Android native SharedPreferences with `MODE_PRIVATE` allows cross-process reads.

```kotlin
private fun configureTunInterface(): ParcelFileDescriptor {
    Log.w(TAG, "configureTunInterface: building TUN...")
    val builder = Builder()
    builder.setMtu(9000)
    builder.addAddress("26.26.26.1", 30)
    builder.addRoute("0.0.0.0", 0)
    builder.addDnsServer("1.1.1.1")
    builder.addDnsServer("8.8.8.8")
    builder.addAddress("da26:2626::1", 126)
    builder.addRoute("::", 0)
    builder.setSession("Arma VPN")

    // Read per-app proxy config (Phase 4, D-04, ROUTE-04)
    val perAppPrefs = applicationContext.getSharedPreferences("per_app_config", MODE_PRIVATE)
    val perAppMode = perAppPrefs.getString("per_app_mode", null)
    val selectedApps = perAppPrefs.getStringSet("selected_apps", emptySet()) ?: emptySet()

    if (perAppMode == "whitelist" && selectedApps.isNotEmpty()) {
        // Whitelist mode: only selected apps route through VPN
        // Do NOT call addDisallowedApplication — can't mix with addAllowedApplication (Pitfall 3)
        // Self-exclusion is implicit: our app is not in the allowed list
        for (pkg in selectedApps) {
            try {
                builder.addAllowedApplication(pkg)
            } catch (e: Exception) {
                Log.w(TAG, "Skipping uninstalled whitelist app: $pkg")
            }
        }
        Log.w(TAG, "Per-app whitelist: ${selectedApps.size} apps allowed")
    } else {
        // Default or blacklist mode: all apps through VPN, exclude selected + self
        builder.addDisallowedApplication(packageName) // Self-exclusion (Pitfall #12)
        if (perAppMode == "blacklist" && selectedApps.isNotEmpty()) {
            for (pkg in selectedApps) {
                if (pkg == packageName) continue // Already excluded
                try {
                    builder.addDisallowedApplication(pkg)
                } catch (e: Exception) {
                    Log.w(TAG, "Skipping uninstalled blacklist app: $pkg")
                }
            }
            Log.w(TAG, "Per-app blacklist: ${selectedApps.size} apps excluded")
        }
    }

    Log.w(TAG, "TUN config: MTU=9000, addr=26.26.26.1/30, route=0.0.0.0/0, DNS=1.1.1.1+8.8.8.8, excl=$packageName")
    val iface = builder.establish()
        ?: throw IllegalStateException("VPN builder.establish() returned null")
    Log.w(TAG, "TUN established: fd=${iface.fd}")
    return iface
}
```

Key design decisions:
- **Whitelist mode**: Uses ONLY `addAllowedApplication()`. Self-exclusion is implicit (our app not in allowed list). No `addDisallowedApplication()` call at all (Pitfall 3 — can't mix).
- **Blacklist mode**: Uses ONLY `addDisallowedApplication()`. Self always excluded. Selected apps also excluded.
- **No per-app config (null mode or empty apps)**: Default behavior — self-exclusion only, all other apps through VPN (existing Phase 2 behavior preserved).
- **Each add call wrapped in try-catch**: Handles uninstalled apps gracefully (Pitfall 2).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Dart → Kotlin MethodChannel | Package names and app icons cross process boundary |
| SharedPreferences → VPN process | Per-app config crosses process boundary via Android SharedPreferences |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-03-01 | Spoofing | Package names in selectedApps | mitigate | Package names come from user selection in app picker (populated by PackageManager). Validate by wrapping addAllowedApplication/addDisallowedApplication in try-catch — invalid names throw NameNotFoundException and are skipped. |
| T-04-03-02 | Tampering | per_app_config SharedPreferences | accept | SharedPreferences is app-private on Android. Only our app processes can read/write. Root access could tamper, but root access bypasses all Android security. |
| T-04-03-03 | Information Disclosure | App icon base64 data | accept | App icons are not sensitive data. They are visible to any app via PackageManager. No privacy concern. |
| T-04-03-04 | Denial of Service | Large number of apps in whitelist/blacklist | accept | Android VpnService.Builder handles large app lists internally. No known DoS vector. Performance may degrade with 500+ apps but this is an edge case. |
| T-04-03-05 | Elevation of Privilege | VPN routing bypass | mitigate | Self-exclusion always applied: in blacklist mode explicitly via addDisallowedApplication(packageName), in whitelist mode implicitly by not adding our app to allowed list. This prevents routing loops (Pitfall #12). |
</threat_model>
